#### Task 1.4 — `prefers-reduced-motion` strategy [F-static-09, M]

- **Why:** P1 accessibility gap per roadmap §2. macOS Reduce-motion users currently get full graph physics and canvas tween animation.
- **Approach:**
  1. Create `src/renderer/src/hooks/useReducedMotion.ts`:
     ```ts
     import { useEffect, useState } from 'react'
     export function useReducedMotion(): boolean {
       const [reduced, setReduced] = useState(() =>
         typeof window !== 'undefined' &&
         window.matchMedia('(prefers-reduced-motion: reduce)').matches
       )
       useEffect(() => {
         const mq = window.matchMedia('(prefers-reduced-motion: reduce)')
         const handler = (e: MediaQueryListEvent) => setReduced(e.matches)
         mq.addEventListener('change', handler)
         return () => mq.removeEventListener('change', handler)
       }, [])
       return reduced
     }
     ```
  2. In `assets/index.css`, add a global CSS fallback:
     ```css
     @media (prefers-reduced-motion: reduce) {
       *, *::before, *::after {
         animation-duration: 0.01ms !important;
         animation-iteration-count: 1 !important;
         transition-duration: 0.01ms !important;
         scroll-behavior: auto !important;
       }
     }
     ```
  3. In Pixi graph physics and canvas tween sites, use `useReducedMotion()` to skip or flatten animations. Grep for `requestAnimationFrame`, `d3-force.tick`, `gsap`, or any animation calls in `src/renderer/src/workers/` and `panels/canvas/`.
- **Definition of done:**
  - `useReducedMotion` hook exists.
  - Global CSS override is in place.
  - At least the graph force settling and canvas tween sites check the hook.
  - macOS Settings → Accessibility → Reduce motion enabled → graph and canvas animations flatten.

#### Task 2.4 — `CardLodPreview` text → `colors.text.secondary` [F-static-04, S]

- **Files:** `src/renderer/src/panels/canvas/CardLodPreview.tsx:14`, plus a new `src/renderer/src/design/theme-vars.ts`
- **Evidence:** `text: '#94a3b8'`
- **Change:** consumers outside Pixi can use `'var(--color-text-secondary)'` directly. Pixi consumers (like `CardLodPreview`) need a resolved hex, so:
  1. Create `src/renderer/src/design/theme-vars.ts`:
     ```ts
     import { useSettingsStore } from '@renderer/store/settings-store'

     const cache = new Map<string, string>()
     let primed = false

     function read(name: string): string {
       const v = getComputedStyle(document.documentElement).getPropertyValue(name).trim()
       cache.set(name, v)
       return v
     }

     export function getThemeVar(name: string): string {
       if (!primed) {
         // Invalidate the cache whenever theme or accent changes.
         useSettingsStore.subscribe(
           (s) => [s.theme, s.accent] as const,
           () => cache.clear()
         )
         primed = true
       }
       const hit = cache.get(name)
       return hit ?? read(name)
     }
     ```
     (Selector arg names match the store; adjust if the store uses different keys.)
  2. In `CardLodPreview.tsx`, replace `text: '#94a3b8'` with `text: getThemeVar('--color-text-secondary')`.
  3. Document the pattern at the top of `theme-vars.ts` — "use from Pixi and other canvas consumers; React components should read `var(--…)` directly."
- **Definition of done:**
